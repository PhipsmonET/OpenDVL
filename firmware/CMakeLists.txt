cmake_minimum_required(VERSION 3.20)

# ==============================================================================
# OpenDVL STM32H7 Firmware Build System
# ==============================================================================
project(OpenDVL_Firmware C CXX ASM)

# Ensure release builds are optimized and debug builds contain debug info
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug" CACHE STRING "Choose the type of build" FORCE)
endif()

# Global definitions for STM32H7
add_definitions(
    -DSTM32H743xx
    -DUSE_HAL_DRIVER
    -DARM_MATH_CM7
)

# Compiler-specific optimization and debug flags
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    add_compile_options(-O0 -g3 -gdwarf-2)
else()
    add_compile_options(-O3)
endif()

# Linker script configuration
# (Typically generated by CubeMX or manual linker script STM32H743VITx_FLASH.ld)
set(LINKER_SCRIPT "${CMAKE_CURRENT_SOURCE_DIR}/STM32H743VITx_FLASH.ld")
add_link_options(-T ${LINKER_SCRIPT})

# ------------------------------------------------------------------------------
# Include Directories
# ------------------------------------------------------------------------------
include_directories(
    Core/Inc
    App/Inc
    Drivers/STM32H7xx_HAL_Driver/Inc
    Drivers/STM32H7xx_HAL_Driver/Inc/Legacy
    Drivers/CMSIS/Device/ST/STM32H7xx/Include
    Drivers/CMSIS/Include
    Drivers/CMSIS/DSP/Include
)

# ------------------------------------------------------------------------------
# Source Files Listing
# ------------------------------------------------------------------------------
file(GLOB_RECURSE SOURCES
    "Core/Src/*.c"
    "App/Src/*.c"
    "App/Src/*.cpp"
    "Drivers/STM32H7xx_HAL_Driver/Src/*.c"
)

# Startup file (ASM)
# Usually provided by ST in Drivers/CMSIS/Device/ST/STM32H7xx/Source/Templates/gcc/startup_stm32h743xx.s
set(STARTUP_SOURCE "Core/Startup/startup_stm32h743xx.s")

# ------------------------------------------------------------------------------
# Build Executable Target
# ------------------------------------------------------------------------------
add_executable(${PROJECT_NAME}.elf ${SOURCES} ${STARTUP_SOURCE})

# Link libraries (e.g. CMSIS-DSP library for FFT functions)
# Target link libraries can be added here as needed:
# target_link_libraries(${PROJECT_NAME}.elf arm_cortexM7lfsp_math)

# ------------------------------------------------------------------------------
# Post-Build Steps (Generate binary/hex files and display memory sizes)
# ------------------------------------------------------------------------------
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O ihex ${PROJECT_NAME}.elf ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_OBJCOPY} -O binary ${PROJECT_NAME}.elf ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_SIZE} --format=berkeley ${PROJECT_NAME}.elf
    COMMENT "Generating auxiliary hex/bin formats and displaying memory usage:"
)
